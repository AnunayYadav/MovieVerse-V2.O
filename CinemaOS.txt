Event
Type
Description

Video started playing.

PLAYER_EVENT
Video was paused.

PLAYER_EVENT
User seeked to a different timestamp.
ended
PLAYER_EVENT
Playback reached the end.

PLAYER_EVENT
Emitted periodically during playback.
playerstatus
PLAYER_EVENT
Response to a getStatus command.
Event listener

javascript
Copy
window.addEventListener('message', ({ origin, data }) => {
  const allowed = ['https://cinemaos.tech'];
  if (!allowed.includes(origin) || !data) return;

  if (data.type === 'PLAYER_EVENT') {
    const { event, currentTime, duration, playing } = data.data;
    console.log(`[${event}] ${currentTime}s / ${duration}s playing=${playing}`);
  }

  if (data.type === 'MEDIA_DATA') {
    localStorage.setItem('cinemaosProgress', JSON.stringify(data.data));
  }
});
PLAYER_EVENT data structure

json
Copy
{
  type: "PLAYER_EVENT",
  data: {
    event: "play" | "pause" | "seeked" | "ended" | "timeupdate" | "playerstatus",
    currentTime: number,
    duration: number,
    tmdbId: string,
    mediaType: "movie" | "tv",
    season?: number,
    episode?: number,
    playing: boolean,
    muted: boolean,
    volume: number
  }
}

Description

—

—
Pause video playback.

time: number

level: 0.0–1.0
Set volume level.

muted: boolean
Mute or unmute the player.

—
Request a playerstatus event back.
Commands example

javascript
Copy
const iframe = document.querySelector('iframe');

// Play / Pause
iframe.contentWindow.postMessage({ command: 'play' }, '*');
iframe.contentWindow.postMessage({ command: 'pause' }, '*');

// Seek to 2 minutes
iframe.contentWindow.postMessage({ command: 'seek', time: 120 }, '*');

// Set volume to 50%
iframe.contentWindow.postMessage({ command: 'volume', level: 0.5 }, '*');

// Mute / Unmute
iframe.contentWindow.postMessage({ command: 'mute', muted: true }, '*');

// Get current player status
iframe.contentWindow.postMessage({ command: 'getStatus' }, '*');
